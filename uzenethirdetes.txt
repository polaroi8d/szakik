CREATE TABLE Hirdetes
(
 H_ID NUMBER(4) NOT NULL,
 SZ_ID NUMBER(4) NOT NULL,
 Datum DATE,
 Szoveg VARCHAR2(500),
 CONSTRAINT ERTEKELES_FOREIGN_KEY FOREIGN KEY (SZ_ID) REFERENCES SZAKI(SZ_ID),
 CONSTRAINT ERTEKELES_PRIMARY_KEY PRIMARY KEY (H_ID, SZ_ID, Datum)
);


CREATE TABLE Uzenet
(
 U_ID NUMBER(4) NOT NULL,
 SZ_ID NUMBER(4) NOT NULL,
  F_ID NUMBER(4) NOT NULL,
 Datum DATE,
 Szoveg VARCHAR2(500),
 CONSTRAINT ERTEKELES_FOREIGN_KEY FOREIGN KEY (SZ_ID) REFERENCES SZAKI(SZ_ID),
 CONSTRAINT ERTEKELES_FOREIGN_KEY2 FOREIGN KEY (F_ID) REFERENCES FELHASZNALO(F_ID),
 CONSTRAINT ERTEKELES_PRIMARY_KEY PRIMARY KEY (H_ID, SZ_ID, Datum, F_ID)
);
